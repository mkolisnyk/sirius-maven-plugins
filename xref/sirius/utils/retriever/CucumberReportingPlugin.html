200     /* (non-Javadoc)
201      * @see org.apache.maven.reporting.AbstractMavenReport#executeReport(java.util.Locale)
202      */
203     @Override
204     protected void executeReport(Locale arg0) throws MavenReportException {
205         File reportOutputDirectory = new File(outputDirectory + fs + "cucumber" );
206         List<String> jsonReportFiles = new ArrayList<String>();
207         jsonReportFiles.add(this.jsonFile);
208 
209         ReportBuilder reportBuilder;
210         try {
211             reportBuilder = 
212                     new ReportBuilder(
213                             jsonReportFiles, 
214                             reportOutputDirectory, 
215                             "", 
216                             project.getVersion(), 
217                             project.getName(), 
218                             skippedFails, 
219                             undefinedFails, 
220                             flashCharts, 
221                             runWithJenkins, 
222                             artifactsEnabled, 
223                             "", 
224                             highCharts);            
225             reportBuilder.generateReports();
226         } catch (Exception e) {
227             // TODO Auto-generated catch block
228             e.printStackTrace();
229         }
230         
231         Sink sink = getSink();
232         sink.head();
233         sink.title();
234         sink.text("Cucumber Test Results Report");
235         sink.title_();
236         sink.head_();
237      
238         sink.body();
239         sink.section1();
240         sink.sectionTitle1();
241         sink.text("Features Overview");
242         sink.sectionTitle1_();
243         sink.paragraph();
244         sink.text("Features overview report shows test results distributed by each specific feature with detailed information about tests included");
245         sink.link("cucumber/feature-overview.html");
246         sink.text("View Report");
247         sink.link_();
248         sink.paragraph_();
249         sink.section1_();
250         
251         sink.section1();
252         sink.sectionTitle1();
253         sink.text("Tags Overview");
254         sink.sectionTitle1_();
255         sink.paragraph();
256         sink.text("Tags overview report shows test results distributed by tag with detailed information about tests included");
257         sink.link("cucumber/tag-overview.html");
258         sink.text("View Report");
259         sink.link_();
260         sink.paragraph_();
261         sink.section1_();
262 
263         sink.body_();
264         sink.flush();
265         sink.close();
266 
267     }
