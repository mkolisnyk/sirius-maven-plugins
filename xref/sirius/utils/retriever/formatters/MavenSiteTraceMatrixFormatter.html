71      public String GetIssue(GHIssue issue) {
72          GHUser user = issue.getAssignee();
73          String output = "";
74  
75          if(!newMilestone){
76              output = "<tr>" + output;
77          }
78          else {
79              GHMilestone milestone = issue.getMilestone();
80              int issuesCount = milestone.getClosedIssues() + milestone.getOpenIssues();
81              int rowSpan = issuesCount;
82              
83              if(rowSpan==0){
84                  rowSpan=1;
85              }
86              
87              output += "<tr><td rowspan=\"" + rowSpan + "\"><a href=\"" + milestone.getUrl()+ "\">" + 
88                  milestone.getTitle() + "</a></td>";
89              
90              if(issuesCount == 0){
91                  output += "<td>N/A</td><td>N/A</td><td>N/A</td></tr>";
92              }
93          }        
94          output += "<td><a href=\"" + issue.getUrl() + "\">" + issue.getTitle() + "</td>" +
95              "<td>"+ (issue.getState().compareTo(GHIssueState.CLOSED) == 0 ? "Yes" : "No" ) + "</td>";
96          if(user != null){
97              try {
98                  output += "<td><a href=\"mailto://" + user.getEmail()+ "\">" + user.getName() + "</a></td>" +
99                  		"<td>" +
100                 		"<img src=\"" + user.getAvatarUrl() + "\" />" +
101                 		"</td></tr>";
102             } catch (IOException e) {
103                 // TODO Auto-generated catch block
104                 e.printStackTrace();
105             }
106         }
107         
108         newMilestone = false;
109         return output;
110     }
